  # React Dateslot-Picker

  **A React date and timeslot picker component can be used to create a 
  interface for booking reservations. Users can select a date from the calendar 
  and then pick a specific timeslot for their reservation.**

  [Github](https://github.com/Francoshum95/react-dateslot-picker)

  ## Install
  ```
  npm install @react/dateslot-picker
  ```
  ```
  yarn add @react/dateslot-picker 
  ```
  
  ## Example

  <DateSlotPicker
    startDate={1672549200000}
    endDate={1701406800000}
    currentDate={1680321600000}
    dailyTimePair={[{
      startTime: [10, 0],

    }, {
      startTime: [14, 0],
      endTime: [23, 59]
    }]}
    disableWeekly={[1]}
    disableSpecific={[1]}
    fullBooking={[1680804000000]}
    timezone={"America/Toronto"}
    disableDate={[1680580800000]}
    duration={30}
    onSelectDatetime={(timestamp) => {console.log(timestamp)}}
  />

    ``` js filename="example.ts"
    import {DateSlotPicker} from '@react/dateslot-picker';
    import '@react/dateslot-picker/dist/style.css'

    const Example  = () => {

      const props = {
        startDate: 1672549200000,
        endDate: 1701406800000,
        currentDate: 1680321600000,
        dailyTimePair: [{
          startTime: [10, 0],
          endTime: [12, 0]
        }, {
          startTime: [14, 0],
          endTime: [23, 59]
        }],
        disableWeekly:[1],
        disableSpecific: [1],
        fullBooking: [1680804000000],
        timezone: "America/Toronto",
        disableDate: [1680580800000],
        duration: 30,
        onSelectDatetime: (timestamp: number) => {console.log(timestamp)}
      };

      return (
        <DateSlotPicker
          {...props}
        />
      )
    }; 
    ```
